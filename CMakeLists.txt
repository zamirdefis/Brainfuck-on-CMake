cmake_minimum_required(VERSION 4.3.2)

project("interpreter")

include("${CMAKE_CURRENT_LIST_DIR}/utils.cmake")

function(check_closures_ code_name rtn)
  string(LENGTH ${${code_name}} max_index)
  dec(max_index)
  set(layer 0)
  foreach(index RANGE 0 ${max_index})
    string(SUBSTRING ${${code_name}} ${index} 1 el) 
    if("${el}" STREQUAL "[")
      inc(layer)
    elseif("${el}" STREQUAL "]")
      dec(layer)
      if(layer LESS 0)
        set(${rtn} FALSE PARENT_SCOPE)
        return()
      endif()
    endif()
  endforeach()
  if(layer)
    set(${rtn} FALSE PARENT_SCOPE)
    return()
  endif()
  set(${rtn} TRUE PARENT_SCOPE)
endfunction()

function(interpret memory_name code_name rtn)
  check_closures_(${code_name} cc_rtn)
  
  if(NOT ${cc_rtn})
    message(SEND_ERROR "Invalid Closures!")
    set(${rtn} ${cc_rtn} PARENT_SCOPE)
    return()
  endif()

  set(closures_stack)
  set(mem_offset 0)
  set(code_offset 0)
  string(LENGTH "${${code_name}}" code_end)

  while(TRUE)
    if(${code_offset} EQUAL ${code_end}) 
      break()
    endif()

    string(SUBSTRING "${${code_name}}" ${code_offset} 1 instruction)
    list(GET ${memory_name} ${mem_offset} data)
    
    if("${instruction}" STREQUAL "+")
      inc(data)
      if(${data} GREATER 255)
        set(data 0)
      endif()
      list_set_at(${memory_name} ${mem_offset} ${data})
    #=====
    elseif("${instruction}" STREQUAL "-")
      dec(data)
      if(${data} LESS 0)
        set(data 255)
      endif()
      list_set_at(${memory_name} ${mem_offset} ${data})
    #=====
    elseif("${instruction}" STREQUAL ">")
      inc(mem_offset)
    #=====
    elseif("${instruction}" STREQUAL "<")
      dec(mem_offset)
    #=====
    elseif("${instruction}" STREQUAL ".")
      message("?")
    #=====
    elseif("${instruction}" STREQUAL ",")
      message("...")
    #=====
    elseif("${instruction}" STREQUAL "[")
      if(NOT data)
        set(layer 0)
        while(true)
          inc(code_offset)
          string(SUBSTRING "${${code_name}}" ${code_offset} 1 instruction)
          if("${instruction}" STREQUAL "[")
            inc(layer)
          elseif("${instruction}" STREQUAL "]")
            if(${layer} EQUAL 0)
              break()
            endif()
            dec(layer)
          endif()
        endwhile()
      else()
        stack_push(closures_stack ${code_offset})
      endif()
    #=====
    elseif("${instruction}" STREQUAL "]")
      if(data)
        stack_top(closures_stack code_offset)
      else()
        stack_pop(closures_stack)
      endif()
    endif()

    inc(code_offset)
  endwhile()

  set(${memory_name} "${${memory_name}}" PARENT_SCOPE)
  set(${rtn} TRUE PARENT_SCOPE)
endfunction()

function(mem_view memory_name size)
  set(str_mem "|")
  dec(size)
  foreach(index RANGE ${size})
    list(GET ${memory_name} ${index} el)
    string(APPEND str_mem "${el}|") 
  endforeach()
  message("${str_mem}")
endfunction()

function(entry)

  set(code "->++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++")
  malloc(200 memory)
  interpret(memory code rtn)
  mem_view(memory 10)

  if(NOT ${rtn})
    message(FATAL_ERROR "Error during interpretation")
  endif()

endfunction()

entry()
