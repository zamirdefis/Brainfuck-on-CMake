set(break_point_char "B") # Breakpoint symbol in the code (does not work unless the "debug" parameter is specified)
set(max_code_view 20)

# code => The code that the interpreter must execute
# source (path to file) File with code (if this parameter is specified, the "code" parameter will be ignored
# input => A buffer for user input. If its size is less than the number of input requests, the program will return an error
# size => Number of cells
# memory_view => -Dmemory_view == dynamic; -Dmemory_view=123 == static (Show only 123 cells); -Dmemory_view=OFF == Disables the memory view window
# code_view => ON/OFF | With debugging enabled, the output is sequential
# cell_trace => Enables highlighting of the last 4 cells
# debug => Enables breakpoints | "debug" can be set to "drop" so that the first breakpoint completely halts execution
#   && If memory_view is undefined, it sets it to "dynamic"
#   && If code_view is not defined, it sets it to "ON"
#   && If cell_trace is not defined, it sets it to "ON

foreach(var_name code source input size memory_view code_view debug cell_trace)
  if(DEFINED ${var_name})
    set(${var_name} "${${var_name}}")
    unset(${var_name} CACHE)
  endif()
endforeach()


if(DEFINED debug)
  if(NOT (DEFINED code_view))
    set(code_view ON)
  endif()
  if(NOT (DEFINED memory_view))
    set(memory_view "dynamic")
  endif()
  if(NOT (DEFINED cell_trace))
    set(cell_trace ON)
  endif()
endif()

if (DEFINED source)
  file(READ "${source}" code)
else()
  if (NOT (DEFINED code))
    set(code "+++++ +++++ [ > +++++ ++ > +++++ +++++ > +++ > + <<<< - ] >> ++++ . ---- < ++  > + . +++++ ++ . . < +++++++ . > +++++++++++ . < . > ----- . ------ . -------- .")
  endif()
endif()

if (DEFINED size)
  malloc(${size} memory)
else()
  malloc(256 memory)
endif()

function(check_closures_ code_name rtn)
  string(LENGTH ${${code_name}} max_index)
  dec(max_index)
  set(layer 0)
  foreach(index RANGE 0 ${max_index})
    string(SUBSTRING ${${code_name}} ${index} 1 el) 
    if("${el}" STREQUAL "[")
      inc(layer)
    elseif("${el}" STREQUAL "]")
      dec(layer)
      if(layer LESS 0)
        set(${rtn} FALSE PARENT_SCOPE)
        return()
      endif()
    endif()
  endforeach()
  if(layer)
    set(${rtn} FALSE PARENT_SCOPE)
    return()
  endif()
  set(${rtn} TRUE PARENT_SCOPE)
endfunction()

set(cell_trace_color_pack "fg_h_black" "fg_cyan" "fg_green" "fg_yellow")

# I already have no idea how this spaghetti code works! Why didn’t I do this using regex? - Because colors break everything >:[
function(set_cell_colors cell_indexes_name str_mem_name color_pack_name cell_offset)
  string(REGEX REPLACE "^\\|" "" clean_str "${${str_mem_name}}")
  string(REGEX REPLACE "\\|$" "" clean_str "${clean_str}")
  string(REPLACE "|" ";" cell_list "${clean_str}")

  list(LENGTH cell_list cell_list_len)
  list(LENGTH ${cell_indexes_name} cell_indexes_len)

  set(${str_mem_name} "|")

  dec(cell_indexes_len)
  dec(cell_list_len)
  list(LENGTH ${color_pack_name} color_pack_len)

  foreach(index RANGE ${cell_list_len})
    list(GET cell_list ${index} el)
    set(colored FALSE)
    if(${index} EQUAL ${cell_offset})
      string(APPEND ${str_mem_name} "${fg_red}${el}${nc}|")
      if(${cell_indexes_len} EQUAL 0)
        return()
      endif()
      continue()
    endif()
    foreach(stack_index RANGE ${cell_indexes_len} 0 -1)
      list(GET ${cell_indexes_name} ${stack_index} cell_index)
      if(${index} EQUAL ${cell_index})
        math(EXPR color_index "${color_pack_len} - 1 - ${cell_indexes_len} + ${stack_index}")
        list(GET ${color_pack_name} ${color_index} color_name)
        string(APPEND ${str_mem_name} "${${color_name}}${el}${nc}|")
        set(colored TRUE)
        break()
      endif()
    endforeach()
    if(NOT colored)
      string(APPEND ${str_mem_name} "${el}|")
    endif()
  endforeach()
  set(${str_mem_name} "${${str_mem_name}}" PARENT_SCOPE)
endfunction()

function(func_memory_view memory_name size cell_offset cell_trace_stack_name)
  set(str_mem "|")
  dec(size)
  foreach(index RANGE ${size})
    list(GET ${memory_name} ${index} el)
    if(${index} EQUAL ${cell_offset})
      if(cell_trace) # workaround (crutch) X_X
        string(APPEND str_mem "> ${el} <|") 
      else()
        string(APPEND str_mem "${fg_h_red}> ${el} <${nc}|") 
      endif()
      continue()
    endif()
    string(APPEND str_mem "${el}|") 
  endforeach()
  if(cell_trace)
    set_cell_colors(${cell_trace_stack_name} str_mem cell_trace_color_pack ${cell_offset})
  endif()
  message("${str_mem}")
endfunction()

macro(macro_memory_view)
  if((DEFINED memory_view) AND memory_view)
    set(size 0) 
    if("${memory_view}" STREQUAL "dynamic")
      set(size ${max_mem_offset})
      inc(size)
    else()
      set(size ${memory_view})
    endif()
    func_memory_view(${memory_name} ${size} ${mem_offset} cell_trace_stack)
  endif()
endmacro()

macro(macro_code_view)
  if(code_view)
    string(SUBSTRING ${${code_name}} 0 ${char_pos} code_part)
    string(LENGTH "${code_part}" code_part_len)

    set(more "")
    if(${code_part_len} GREATER ${max_code_view})
      math(EXPR sep_index "${code_part_len} - ${max_code_view}")
      string(SUBSTRING ${code_part} ${sep_index} ${max_code_view} code_part)
      set(more "&&& ")
    endif()
    
    message("${more}${code_part} (<= ${char_pos} char)")
  endif()
endmacro()

macro(interpreter_rtn)
  message("${output_buffer}") 
  set(${memory_name} "${${memory_name}}" PARENT_SCOPE)
  set(${last_cell_name} ${mem_offset} PARENT_SCOPE)
  
  macro_memory_view()
  macro_code_view()

  set(${rtn} TRUE PARENT_SCOPE)
  return()
endmacro()

function(interpret memory_name code_name last_cell_name rtn)
  check_closures_(${code_name} cc_rtn)

  set(cell_trace_stack "")
  
  if(NOT ${cc_rtn})
    message(SEND_ERROR "Invalid Closures!")
    set(${rtn} ${cc_rtn} PARENT_SCOPE)
    return()
  endif()

  set(input_buffer "")
  if(DEFINED input)
    set(input_buffer "${input}") 
  endif()
  set(input_offset 0)

  set(closures_stack)
  set(mem_offset 0)
  set(max_mem_offset 0)
  set(previous_mem_offset 0)
  set(code_offset 0)
  set(char_pos 0)
  string(LENGTH "${${code_name}}" code_end)
  set(output_buffer "")

  while(TRUE)
    if(${code_offset} EQUAL ${code_end}) 
      break()
    endif()

    set(previous_mem_offset ${mem_offset})

    string(SUBSTRING "${${code_name}}" ${code_offset} 1 instruction)
    list(GET ${memory_name} ${mem_offset} data)
    
    if("${instruction}" STREQUAL "+")
      inc(data)
      if(${data} GREATER 255)
        set(data 0)
      endif()
      list_set_at(${memory_name} ${mem_offset} ${data})
    #=====
    elseif("${instruction}" STREQUAL "-")
      dec(data)
      if(${data} LESS 0)
        set(data 255)
      endif()
      list_set_at(${memory_name} ${mem_offset} ${data})
    #=====
    elseif("${instruction}" STREQUAL ">")
      inc(mem_offset)
      if(${mem_offset} GREATER ${max_mem_offset})
        set(max_mem_offset ${mem_offset})
      endif()
    #=====
    elseif("${instruction}" STREQUAL "<")
      dec(mem_offset)
    #=====
    elseif("${instruction}" STREQUAL ".")
      if(${data} EQUAL 12) # FLUSH >_<
        message("${output_buffer}") 
        set(output_buffer "")
      elseif(NOT (${data} EQUAL 0))
        string(ASCII ${data} char)
        string(APPEND output_buffer ${char})
      endif()
    #=====
    elseif("${instruction}" STREQUAL ",")
      if("${input_buffer}" STREQUAL "")
        message(SEND_ERROR "Input overflow")
        set(${rtn} TRUE PARENT_SCOPE)
        return()
      else()
        string(LENGTH ${input_buffer} input_buffer_len)
        if(input_buffer_len LESS_EQUAL input_offset)
          message(SEND_ERROR "Input overflow")
          set(${rtn} TRUE PARENT_SCOPE)
          return()
        endif()
      endif()
      set(char_data "")
      string(SUBSTRING "${input_buffer}" ${input_offset} 1 char_data)
      char_to_ascii(char_data data)
      list_set_at(${memory_name} ${mem_offset} ${data})
      inc(input_offset)
    #=====
    elseif("${instruction}" STREQUAL "[")
      if(NOT data)
        set(layer 0)
        while(TRUE)
          inc(code_offset)
          string(SUBSTRING "${${code_name}}" ${code_offset} 1 instruction)
          if("${instruction}" STREQUAL "[")
            inc(layer)
          elseif("${instruction}" STREQUAL "]")
            if(${layer} EQUAL 0)
              break()
            endif()
            dec(layer)
          endif()
        endwhile()
      else()
        stack_push(closures_stack ${code_offset})
      endif()
    #=====
    elseif("${instruction}" STREQUAL "]")
      if(data)
        stack_top(closures_stack code_offset)
      else()
        stack_pop(closures_stack)
      endif()
    #=====
    elseif(("${instruction}" STREQUAL ${break_point_char}) AND (DEFINED debug))
      if(${debug} STREQUAL "drop")
        interpreter_rtn()
      endif()
      macro_memory_view()
      inc(char_pos)
      macro_code_view()
      dec(char_pos)
    endif()

    if(cell_trace)
      if(NOT (${previous_mem_offset} EQUAL ${mem_offset}))
        list(LENGTH cell_trace_stack cell_trace_stack_len)
        if(${cell_trace_stack_len} GREATER 3)
          list(REMOVE_AT cell_trace_stack 0)
        endif()
        list(APPEND cell_trace_stack ${previous_mem_offset})
      endif()
    endif()
    inc(code_offset)
    set(char_pos ${code_offset})
  endwhile()

  dec(char_pos)
  interpreter_rtn()
endfunction()

function(entry)
  
  interpret(memory code last_cell rtn)
  # mem_view(memory 10)

  if(NOT ${rtn})
    message(FATAL_ERROR "Error during interpretation")
  endif()

endfunction()
